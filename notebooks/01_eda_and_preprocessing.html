# Leakage-safe preprocessing Pipeline -- fit ONLY on X_train, never on the full df
from sklearn.compose import ColumnTransformer
from sklearn.pipeline import Pipeline
from sklearn.impute import SimpleImputer
from sklearn.preprocessing import OneHotEncoder, StandardScaler

# Numeric pipeline: fill missing values with median, then scale
numeric_transformer = Pipeline(steps=[
    ("imputer", SimpleImputer(strategy="median")),
    ("scaler", StandardScaler())
])

# Categorical pipeline: fill missing values with the most frequent category, then one-hot encode
categorical_transformer = Pipeline(steps=[
    ("imputer", SimpleImputer(strategy="most_frequent")),
    ("encoder", OneHotEncoder(handle_unknown="ignore"))
])

# Combine both into one preprocessor
preprocessor = ColumnTransformer(transformers=[
    ("num", numeric_transformer, numeric_cols),
    ("cat", categorical_transformer, categorical_cols)
])

print("Preprocessor built. It will be fit only on X_train when used inside a model Pipeline.")
Preprocessor built. It will be fit only on X_train when used inside a model Pipeline.

# Overfitting - 
#Where the Decision Tree overfits:
#Test R² peaks at max_depth≈5 (~0.865) and then gets worse as the tree is allowed to grow deeper, 
#while train R² keeps climbing all the way to a perfect 1.0. This means the tree starts overfitting beyond depth≈5 — past that point, it's no longer learning real price patterns, it's just memorizing individual training houses. The unrestricted tree (no depth limit) confirms this: train R²=1.0 vs. test R²=0.83 for regression, and train accuracy=1.0 vs. test accuracy=0.9249 for classification — both gaps are clear, measurable evidence of overfitting.

from sklearn.model_selection import validation_curve

# We sweep max_depth (the tree's complexity knob) and watch train vs test R²
# at each setting -- this shows us exactly where the tree starts memorizing
# training data instead of learning a generalizable pattern (overfitting).
depths = [2, 3, 5, 7, 10, 15, 20, None]
train_scores, test_scores = [], []

for d in depths:
    # Build a fresh tree at this depth, wrapped in our same leakage-safe Pipeline
    tree = Pipeline(steps=[
        ("preprocessor", preprocessor),
        ("model", DecisionTreeRegressor(max_depth=d, random_state=RANDOM_STATE))
    ])
    tree.fit(X_train, y_train_reg)

    # Score on training data (what the tree has seen) and test data (what it hasn't)
    train_scores.append(r2_score(y_train_reg, tree.predict(X_train)))
    test_scores.append(r2_score(y_test_reg, tree.predict(X_test)))

x_labels = [str(d) if d else "None" for d in depths]  # "None" = no depth limit at all

# Plot both curves together -- the GAP between them is the overfitting signal
plt.figure(figsize=(9,5))
plt.plot(x_labels, train_scores, "b-o", label="Train R²")
plt.plot(x_labels, test_scores, "r-o", label="Test R²")
plt.xlabel("max_depth")
plt.ylabel("R²")
plt.title("Decision Tree: Train vs Test R² by Depth (Overfitting Curve)")
plt.legend()
plt.grid(True, alpha=0.3)
plt.show()
